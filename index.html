  1. Problem: You are given a string (STR) of length N, consisting of only the lower case English alphabet. Your task is to remove all the duplicate occurrences of characters in the string.

    2. Answer:

            function removeDuplicate(STR,N){
    var index = 0;
    for (var i = 0; i < N; i++){
        var j;
            for (j = 0; j < i; j++)
            {
                if (STR[i] == STR[j])
                {
                    break;
                }
            }
            if (j == i)
            {
                STR[index++] = STR[i];
            }
        }
    
    return STR.join("").slice(STR, index);
    
}
